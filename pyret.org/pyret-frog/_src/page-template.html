data BinTree:
  | leaf
  | node(value, left, right)
end

fun tree-sum(t):
  doc: "Calculate the sum of node values"
  cases(BinTree) t:
    | leaf => 0
    | node(v, l, r) =>
      v + tree-sum(l) + tree-sum(r)
  end
where:
  tree-sum(leaf) is 0
  node4 = node(4, leaf, leaf)
  tree-sum(node(5, node4, leaf)) is 9
end
}

@disable-prefix{
# Note the refinements on the
# l argument and the result.
# The refinments are critical to the
# correct behavior of insertion.

fun insert(n :: Number,
           l :: List(is-sorted-ascending))
    -> List(is-sorted-ascending):
  cases (List) l:
    | empty => [n]
    | link(f, r) =>
      if n < f: link(n, l)
      else:     link(f, insert(n, r))
      end
  end
end
}

eps = 0.001
fun d-dx(f):
  doc: "Calculate the derivative of f"
  fun(x): (f(x + eps) - f(x)) / eps;
where:
  fun square(x): x * x;
  fun two-x(x): 2 * x;
  fun within(delta, target):
    fun(actual):
      (actual - target).abs() < delta;
  end

  d-dx(square)(5) satisfies
    within(0.1, two-x(5))
  d-dx(square)(10) satisfies
    within(0.1, two-x(10))
end
